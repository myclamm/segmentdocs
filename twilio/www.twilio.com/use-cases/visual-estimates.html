Twilio 101

How to

Simple code to build Visual Estimates onto a flexible and easy Twilio API

from flask import Flask
import twilio.twiml

app = Flask(__name__)

@app.route("/", methods=['GET', 'POST'])
def hello_monkey():
    """Respond to incoming calls with a simple text message."""
    resp = twilio.twiml.Response()
    resp.message("Thanks! We'll get back to you with an estimate.")
    return str(resp)

if __name__ == "__main__":
    app.run(debug=True)
