  process.env.TWILIO_AUTH_TOKEN

client.messages.create({
  from: 'DEINE_TWILIO_NUMMER',
  to: 'DEINE_EIGENE_NUMMER',
  body: 'Du hast gerade deine erste SMS von Node.js mit Twilio verschickt!'
}, function (err, message) {
  if(err) {

  } else {
